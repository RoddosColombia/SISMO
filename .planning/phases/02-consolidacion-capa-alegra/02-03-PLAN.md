class AlegraService:
    def __init__(self, db):
        self.db = db

    async def request(self, endpoint, method="GET", body=None, params=None):
        """GET/POST/PUT/DELETE con manejo de errores, auth, retry-ready."""

    async def request_with_verify(self, endpoint, method, body=None) -> dict:
        """POST + GET verificacion. Lanza HTTPException si verificacion falla."""

    async def retry_request(self, endpoint, method, body=None, max_retries=3, delay_base=2.0) -> dict:
        """request() con reintentos automaticos para 429/503."""

    async def is_demo_mode(self):
        """True si no hay credenciales configuradas."""

    async def get_auth_header(self):
        """Headers con Basic Auth."""
