**Step B: Modify `backend/ai_chat.py` process_chat() — SURGICAL CHANGES ONLY**

The changes go in the block between line ~3812 (after building _system_text, _chat_msgs) and line ~3858 (the return statement). The flow becomes:

```python
# ── Feature flag check (per D-02) ──
tool_use_enabled = os.environ.get("TOOL_USE_ENABLED", "").lower() == "true"

if tool_use_enabled:
    from tool_definitions import get_tool_schemas_for_api, TOOL_DEFS
    tools_param = get_tool_schemas_for_api()
else:
    tools_param = None  # No tools= kwarg → XML flow

# ── LLM call (per D-05: preserve cache_control ephemeral) ──
create_kwargs = dict(
    model="claude-opus-4-6",
    max_tokens=4096,
    system=[{"type": "text", "text": _system_text, "cache_control": {"type": "ephemeral"}}],
    messages=_chat_msgs,
)
if tools_param:
    create_kwargs["tools"] = tools_param

_chat_resp = await _chat_client.messages.create(**create_kwargs)

# ── Tool use detection (per D-01, D-04) ──
if tool_use_enabled and _chat_resp.stop_reason == "tool_use":
    # Extract text blocks and tool_use block
    text_parts = [b.text for b in _chat_resp.content if b.type == "text"]
    tool_block = next((b for b in _chat_resp.content if b.type == "tool_use"), None)

    clean_response = "\n".join(text_parts).strip()
    tool_name = tool_block.name
    tool_input = tool_block.input
    tool_id = tool_block.id

    tool_def = TOOL_DEFS.get(tool_name)

    if tool_def and tool_def["requires_confirmation"]:
        # Per D-01: return pending_action WITHOUT executing
        pending_action = {"type": tool_name, "payload": tool_input, "tool_use_id": tool_id}
        # Save assistant message
        await db.chat_messages.insert_one({...})
        return {"message": clean_response, "pending_action": pending_action, "session_id": session_id}

    elif tool_def and not tool_def["requires_confirmation"]:
        # Per D-01: auto-execute
        from tool_executor import execute_tool
        tool_result = await execute_tool(tool_name, tool_input, db, user)

        # Agentic loop: send tool_result back to model for natural language response
        _chat_msgs.append({"role": "assistant", "content": _chat_resp.content})
        _chat_msgs.append({"role": "user", "content": [
            {"type": "tool_result", "tool_use_id": tool_id, "content": json.dumps(tool_result, ensure_ascii=False, default=str)}
        ]})

        # Second LLM call to get natural language summary
        _followup = await _chat_client.messages.create(**{**create_kwargs, "messages": _chat_msgs})
        followup_text = _followup.content[0].text

        await db.chat_messages.insert_one({...})
        return {"message": followup_text, "pending_action": None, "session_id": session_id, "tool_result": tool_result}

# ── Fallback: XML flow (per D-04 — for non-tool responses or TOOL_USE_ENABLED=false) ──
response_text = _chat_resp.content[0].text
# ... existing XML parsing block unchanged ...
```
