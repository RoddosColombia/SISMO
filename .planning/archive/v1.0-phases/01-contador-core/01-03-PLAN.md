```python
import hashlib
import json

def generar_idempotency_key(operacion: str, entity_type: str, datos: dict) -> str:
    """Genera key deterministica para idempotency check.
    CRITICAL: datos must be sorted canonically to ensure same input = same hash."""
    datos_canonicos = json.dumps(datos, sort_keys=True, ensure_ascii=False)
    hash_datos = hashlib.sha256(datos_canonicos.encode()).hexdigest()[:16]
    return f"{operacion}:{entity_type}:{hash_datos}"
```

```python
async def mover_a_dead_letter(db, event_id: str, event_type: str, payload: dict, error: str):

    })
```

```python
async def _process_dead_letter_queue():
    """Procesa items pendientes en dead_letter_queue. Corre cada 60 segundos."""
    from database import db
    ahora = datetime.now(timezone.utc)
    items = await db.dead_letter_queue.find(
        {"status": "pending", "next_retry": {"$lte": ahora}}
    ).limit(20).to_list(20)

    for item in items:
        try:
            # Re-process the webhook event
            # Import the handler function from alegra_webhooks
            from routers.alegra_webhooks import procesar_evento_webhook
            await procesar_evento_webhook(db, item["event_type"], item["payload"])

            # Success — mark as resolved
            await db.dead_letter_queue.update_one(
                {"_id": item["_id"]},
                {"$set": {"status": "resolved", "resolved_at": datetime.now(timezone.utc)}}
            )
            logger.info(f"DLQ item {item['event_id']} resolved after {item['attempts']} attempts")

        except Exception as e:
            attempts = item["attempts"] + 1
            if attempts > 5:
                # Max retries exhausted — mark as failed, emit alert event
                await db.dead_letter_queue.update_one(
                    {"_id": item["_id"]},
                    {"$set": {"status": "failed", "error": str(e), "attempts": attempts}}
                )
                # Emit dlq.exhausted event per D-04d
                from event_bus import emit_event
                await emit_event(db, "dead_letter_queue", "dlq.exhausted", {
                    "event_id": item["event_id"],
                    "event_type": item["event_type"],
                    "attempts": attempts,
                    "last_error": str(e),
                })
                logger.error(f"DLQ item {item['event_id']} FAILED after {attempts} attempts: {e}")
            else:
                # Schedule next retry with backoff
                backoff_idx = min(attempts - 1, len(BACKOFF_MINUTES) - 1)
                next_retry = datetime.now(timezone.utc) + timedelta(minutes=BACKOFF_MINUTES[backoff_idx])
                await db.dead_letter_queue.update_one(
                    {"_id": item["_id"]},
                    {"$set": {"attempts": attempts, "next_retry": next_retry, "error": str(e)}}
                )
```

```python
def _invalidate_keys(prefixes: list[str]) -> None:
    """Invalida todas las keys de cache que matchean los prefijos dados."""
    global _cache  # or whatever the cache dict is named
```
